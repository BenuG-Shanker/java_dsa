public void removeCycle(){
	Node slow = head;
	Node fast = head;
	boolean cycle = false;
	while(fast != null && fast.next != null){
		slow = slow.next;
		fast = fast.next.next;
		if(slow == fast){
			cycle = false;
			break;
		}
	}
	if(cycle == false){
		return;
	}
	slow = head;
	Node prev = null;
	while(slow != fast){
		slow = slow.next;
		fast = fast.next;
		prev = fast;
	}
	prev.next = null;
}
